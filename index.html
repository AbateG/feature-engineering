Step 9: Cleaning the dataset (handling missing values and outliers)

In[]:
# Drop duplicates
df = df.drop_duplicates().copy()

# Handle missing values safely
df.loc[:, 'BATH'] = df['BATH'].fillna(df['BATH'].median())

# Outlier removal: cap extreme values (top 1%)
price_cap = df['PRICE'].quantile(0.99)
sqft_cap = df['PROPERTYSQFT'].quantile(0.99)

df = df.loc[
    (df['PRICE'] <= price_cap) &
    (df['PROPERTYSQFT'] <= sqft_cap)
].copy()

# Final shape
